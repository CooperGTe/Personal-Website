Connecting To Network

(for device with wifi card and not ethernet)

# iwtcl
# device list
# station [device] scan
# station [device] get-networks
# station [device] connect [ssid]

Partitioning

(the easy way)

# cfdisk

Note: you can check your partition and what u mounting using

# lsblk

Formatting

# mkfs.ext4 [partitition location]
# mkfs.fat -F 32 [boot partition]
# mkswap [swap partition]

Mounting

# mount [partition location] /mnt
# mkdir -p /mnt/boot/efi
# mount [boot partition] /mnt/boot/efi
# swapon [swap partition]

Mirrorlist(rankmirror)

# cp /etc/pacman.d/mirrorlist /etc/pacman.d/mirrorlist.backup
# sudo pacman -Sy pacman-contrib
# rankmirrors -n 6 /etc/pacman.d/mirrorlist.backup > /etc/pacman.d/mirrorlist

Mirrorlist(reflector)

# sudo reflector --country 'indonesia,singapure,' --verbose --protocol https -f 10 --score 20 --save /etc/pacman.d/mirrorlist

Installing The Kernel

# pacstrap /mnt base linux linux-firmware base-devel grub efibootmgr neovim networkmanager

Installing The Kernel(Offline Way)

Generate Fstab

# genfstab /mnt
# genfstab /mnt > /mnt/etc/fstab
# cat /mnt/etc/fstab

Enter The Installed Arch

# arch-chroot /mnt

Timezone Setup

# ln -sf /usr/share/zoneinfo/Asia/Jakarta /etc/localtime
# hwclock --systohc

Locale Setup

# nvim /etc/locale.gen
# en_US.UTF-8 UTF-8
# locale-gen
# nvim /etc/locale.conf
# LANG=en_US.UTF-8

Host name

# nvim /etc/hostname

Root Password

# passwd

Adding User

# useradd -m -G wheel -s /bin/bash [name]
# passwd [name]

Add Group To the User

# EDITOR=nvim visudo
# %wheel ALL=(ALL.ALL) ALL

Enable Core Service

# systemctl enable NetworkManager

Bootloader Installation

# grub-install /dev/sda
# grub-mkconfig -o /boot/grub/grub.cfg

Finish Up

# exit
# umount -a
# reboot
